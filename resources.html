Node.js Implementation

v18+ 
const iot = require('@iot-connect/sdk');

// Initialize client
const client = new iot.Client('YOUR_KEY');

// List active nodes
client.getDevices({ status: 'online' })
  .then(devices => console.log(devices));

Python (MicroPython)

ESP32 
import iot_connect

gateway = iot_connect.Gateway()

def on_trigger(evt):
    print("Event caught:", evt.name)

gateway.subscribe("alert/temp", on_trigger)
